// add class to element
  document.querySelector('elementName').classList.add('.class')
  // remove class to elment
  document.querySelector('elementName').classList.remove('.class')
  // check if the element has this class or not
  document.querySelector('elementName').classList.contain('.class')
  // toggle the class if exist remove , if not add
  document.querySelector('elementName').classList.toggle('.class')
